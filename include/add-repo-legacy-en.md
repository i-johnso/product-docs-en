=== "Debian 8.x (jessie)"
    ``` bash
    apt-key adv --keyserver keys.gnupg.net --recv-keys 72B865FD
    echo 'deb http://repo.wallarm.com/debian/wallarm-node jessie/' >/etc/apt/sources.list.d/wallarm.list
    apt-get update
    ```
